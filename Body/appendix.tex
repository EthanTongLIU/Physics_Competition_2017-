clear,clc,clf;
pi = 3.1415926; 
for k = -12 : -1
    plot( [0.1*pi*k,0.1*pi*k],[0,0.25*sin(2.5*0.1*pi*k)/(2.5*0.1*pi*k)],'r-' );
    hold on;
    plot( 0.1*pi*k,0.25*sin(2.5*0.1*pi*k)/(2.5*0.1*pi*k),'m.','markersize',20 );
    hold on;
    text( 0.1*pi*k,0.25*sin(2.5*0.1*pi*k)/(2.5*0.1*pi*k),['  a_{',num2str(k),'}'] );
    hold on;
end
